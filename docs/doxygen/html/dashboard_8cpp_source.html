36void Dashboard::drawGauge(cv::Mat& frame, cv::Point center, int radius, double value, double max_value,
37 const std::string& label, const std::string& unit, bool is_red_zone) {
38
39 // Фон прибора - темное кольцо
40 cv::circle(frame, center, radius, cv::Scalar(50, 50, 50), 4, cv::LINE_AA);
41
42 if (value > max_value) value = max_value;
43 if (value < 0) value = 0;
44
45 double angle_span = 270.0;
46 double start_angle = 135.0;
47 double fill_angle = (value / max_value) * angle_span;
48
49 // В OpenCV цвета задаются как BGR (Blue, Green, Red)
50 cv::Scalar core_color = is_red_zone ? cv::Scalar(50, 50, 255) : cv::Scalar(255, 230, 50); // Красный / Голубой (Cyan)
51 cv::Scalar glow_color = is_red_zone ? cv::Scalar(0, 0, 150) : cv::Scalar(100, 100, 0);
52
53 if (fill_angle > 0) {
54 // Свечение (Glow effect) - толстая линия темного тона
55 cv::ellipse(frame, center, cv::Size(radius, radius), 0, start_angle, start_angle + fill_angle, glow_color, 12, cv::LINE_AA);
56 // Ядро (Core) - тонкая яркая линия
57 cv::ellipse(frame, center, cv::Size(radius, radius), 0, start_angle, start_angle + fill_angle, core_color, 3, cv::LINE_AA);
58 }
59
60 // Значение по центру
61 std::string val_str = std::to_string(static_cast<int>(value));
62 int baseline = 0;
63 cv::Size text_size = cv::getTextSize(val_str, cv::FONT_HERSHEY_DUPLEX, 0.8, 2, &baseline);
64 cv::Point text_pos(center.x - text_size.width / 2, center.y + 8);
65
66 // Свечение текста
67 cv::putText(frame, val_str, text_pos, cv::FONT_HERSHEY_DUPLEX, 0.8, glow_color, 5, cv::LINE_AA);
68 cv::putText(frame, val_str, text_pos, cv::FONT_HERSHEY_DUPLEX, 0.8, cv::Scalar(255, 255, 255), 2, cv::LINE_AA);
69
70 // Название
71 cv::putText(frame, label, cv::Point(center.x - 24, center.y + radius + 18),
72 cv::FONT_HERSHEY_SIMPLEX, 0.45, cv::Scalar(220, 220, 220), 1, cv::LINE_AA);
73 cv::putText(frame, unit, cv::Point(center.x - 16, center.y + radius + 32),
74 cv::FONT_HERSHEY_SIMPLEX, 0.35, cv::Scalar(130, 130, 130), 1, cv::LINE_AA);
75}

77void Dashboard::drawLinearGauge(cv::Mat& frame, cv::Rect rect, double value, double max_value,

83 if (value > max_value) value = max_value;

96

99

101 std::string val_str = std::to_string(static_cast<int>(value));
102 cv::putText(frame, val_str, cv::Point(rect.x + rect.width - 35, rect.y - 10),
103 cv::FONT_HERSHEY_DUPLEX, 0.5, cv::Scalar(255, 255, 255), 1, cv::LINE_AA);
