21DriverState DMSMonitor::analyze(const cv::Mat& frame) {
22 DriverState state;
23
24 if (frame.empty()) {
25 state.face_detected = false;
26 state.eyes_open = false;
27 state.alert_distracted = true;
28 return state;
29 }
30
31 // 1. Детекция лица (Оптимизация FPS: запускаем нейросеть только 1 раз в 5 кадров)
32 if (frame_counter_ % 5 == 0 || last_face_rect_.area() == 0) {
33 last_face_rect_ = detectFace(frame);
34 }
35 frame_counter_++;
36
37 state.face_rect = last_face_rect_;
38
39 if (state.face_rect.area() > 0) {
40 state.face_detected = true;
41
42 // 2. Детекция угла поворота головы (аппроксимация по смещению от центра кадра)
43 state.head_turn_deg = estimateHeadTurn(frame, state.face_rect);
44 state.looking_forward = std::abs(state.head_turn_deg) < 20.0f; // Допуск 20 градусов
45 state.alert_distracted = !state.looking_forward;
46
47 // 3. Детекция глаз
48 state.eye_openness = estimateEyeOpenness(frame, state.face_rect);
49 state.eyes_open = (state.eye_openness > 0.0f);
50 } else {
51 state.face_detected = false;
52 state.eyes_open = false; // Нет лица -> глаза не видны
53 state.looking_forward = false;
54 state.alert_distracted = true; // Отвлечение внимания, если лица вообще не видно
55 }
56
57 // 4. Обновление истории глаз (Буфер увеличен до 30 кадров для 30 FPS)
58 eye_history_.push_back(state.eyes_open);
59 if (eye_history_.size() > 30) {
60 eye_history_.pop_front();
61 }
62
63 // 5. Определение усталости (закрыты в 20 из 30 последних кадров)
64 int closed_count = 0;
65 for (bool open : eye_history_) {
66 if (!open) closed_count++;
67 }
68
69 if (closed_count >= 20 && eye_history_.size() >= 20) {
70 state.alert_drowsy = true;
71 } else {
72 state.alert_drowsy = false;
73 }
74
75 return state;
76}

79 if (face_net_.empty()) return cv::Rect(0, 0, 0, 0);

83
84 face_net_.setInput(blob);

86

89
90 float max_confidence = 0.0f;

92
93 for (int i = 0; i < detectionMat.rows; i++) {

95
96 if (confidence > 0.5f && confidence > max_confidence) {

98
99 int x1 = static_cast<int>(detectionMat.at<float>(i, 3) * frame.cols);
100 int y1 = static_cast<int>(detectionMat.at<float>(i, 4) * frame.rows);
101 int x2 = static_cast<int>(detectionMat.at<float>(i, 5) * frame.cols);
102 int y2 = static_cast<int>(detectionMat.at<float>(i, 6) * frame.rows);
103
104 // Ограничение координат границами кадра
105 x1 = std::max(0, x1);
106 y1 = std::max(0, y1);
107 x2 = std::min(frame.cols - 1, x2);
108 y2 = std::min(frame.rows - 1, y2);

117float DMSMonitor::estimateEyeOpenness(const cv::Mat& frame, cv::Rect face_rect) {
118 if (eye_cascade_.empty()) return 0.0f;
119
120 // Ищем глаза только в верхней половине лица
121 cv::Rect upper_face = face_rect;
122 upper_face.height = upper_face.height / 2;
123 // Срезаем немного сбоку, чтобы не захватывать уши/волосы (оптимизация)
124 upper_face.x += static_cast<int>(upper_face.width * 0.1);
125 upper_face.width = static_cast<int>(upper_face.width * 0.8);
126 upper_face.y += static_cast<int>(upper_face.height * 0.2);
127
128 // Убедимся, что прямоугольник внутри кадра
129 upper_face &= cv::Rect(0, 0, frame.cols, frame.rows);
130 if(upper_face.area() == 0) return 0.0f;
131
132 cv::Mat faceROI = frame(upper_face);
133 cv::Mat grayROI;
134 cv::cvtColor(faceROI, grayROI, cv::COLOR_BGR2GRAY);
135 cv::equalizeHist(grayROI, grayROI);
136
137 std::vector<cv::Rect> eyes;
138 // Минимальный размер глаз, чтобы отсеять шумы
139 int min_eye_size = std::max(10, static_cast<int>(upper_face.width * 0.15));
140 eye_cascade_.detectMultiScale(grayROI, eyes, 1.1, 3, 0, cv::Size(min_eye_size, min_eye_size));
141
142 // Возвращаем: есть хотя бы 1 глаз -> 1.0 (закрыты -> 0.0)
143 // Каскады Хаара плохо ловят закрытые глаза, поэтому если найдены - открыты.
144 if (eyes.size() > 0) return 1.0f;
145 return 0.0f;
146}

148float DMSMonitor::estimateHeadTurn(const cv::Mat& frame, cv::Rect face_rect) {
149 // Упрощенная логика: смотрим смещение центра лица от центра кадра.
150 // Если камера стоит по центру перед водителем, то сильное смещение влево/вправо
151 // или отсутствие симметрии можно трактовать как отвлечение.
152 float face_center_x = face_rect.x + (face_rect.width / 2.0f);
153 float frame_center_x = frame.cols / 2.0f;
154
155 float offset = (face_center_x - frame_center_x) / frame_center_x; // от -1 до 1
156 // Условная конвертация в градусы (до 45 градусов поворота)
157 return offset * 45.0f;
158}
