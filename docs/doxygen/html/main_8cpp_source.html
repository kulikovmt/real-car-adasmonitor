43void obd_thread_func(SharedState* state) {
44 try {

46
47 std::ifstream csv_file("data/dataset.csv");

52
53 // Пропускаем заголовок
54 std::string header_line;
55 std::getline(csv_file, header_line);
56 std::cout << "[OBD Thread] CSV открыт. Начинаю потоковое чтение...\n";
57
58 OBDParser parser; // Используем только для parseLineSafe
59 std::string line;
60 int line_num = 1;
61 int ok_count = 0;
62
63 while (state->running) {
64 // Читаем следующую строку из CSV
65 if (!std::getline(csv_file, line)) {
66 // Конец файла — возвращаемся к началу
67 csv_file.clear();
68 csv_file.seekg(0);
69 std::getline(csv_file, header_line); // Пропустить заголовок
70 line_num = 1;
71 continue;
72 }
73 line_num++;
74
75 if (line.empty()) continue;
76
77 // Парсим строку
78 OBDRecord rec;
79 // Вручную парсим строку (ищем нужные столбцы)
80 std::vector<std::string> tokens;
81 std::string token;
82 std::stringstream ss(line);
83 while (std::getline(ss, token, ',')) {
84 tokens.push_back(token);
85 }
86
87 if (tokens.size() <= 12) {
88 continue; // Строка слишком короткая
89 }
90
91 try {
92 // Индексы из CSV: 1=RPM, 2=SPEED, 3=THROTTLE_POS, 7=COOLANT_TEMP, 11=ENGINE_LOAD, 12=FUEL_LEVEL
93 rec.speed = tokens[2].empty() ? 0.0 : std::stod(tokens[2]);
94 rec.rpm = tokens[1].empty() ? 0.0 : std::stod(tokens[1]);
95 rec.throttle_pos = tokens[3].empty() ? 0.0 : std::stod(tokens[3]);
96 rec.coolant_temp = tokens.size() > 7 && !tokens[7].empty() ? std::stod(tokens[7]) : 85.0;
97 rec.engine_load = tokens.size() > 11 && !tokens[11].empty() ? std::stod(tokens[11]) : 30.0;
98 rec.fuel_level = tokens.size() > 12 && !tokens[12].empty() ? std::stod(tokens[12]) : 50.0;
99
100 // Стиль вождения (столбец 30)
101 if (tokens.size() > 30 && !tokens[30].empty()) {
102 rec.style = OBDParser::convertLabelToBehavior(tokens[30]);
103 } else {
104 rec.style = BehaviorStyle::NORMAL;
105 }
106 } catch (...) {
107 continue; // Пропускаем битую строку
108 }
109
110 // Результат классификации из CSV-метки
111 ClassificationResult res;
112 res.label = static_cast<int>(rec.style);
113 if (res.label < 0 || res.label > 2) res.label = 1;
114 res.confidence = 1.0f;
115 res.scores = {0, 0, 0};
116 res.scores[res.label] = 1.0f;
117
118 // Обновление SharedState
119 {
120 std::lock_guard<std::mutex> lock(state->mtx);
121 state->current_obd = rec;
122 state->obd_class = res;
123 state->total_obd_records++;
124
125 if (res.label == 2) {
126 state->aggressive_alerts++;
127 }
128 }
129
130 ok_count++;
131 if (ok_count == 1) {
132 std::cout << "[OBD Thread] Первая запись отправлена: speed=" << rec.speed << " rpm=" << rec.rpm << "\n";
133 }
134
135 // 10 Hz = 100 мс
136 std::this_thread::sleep_for(std::chrono::milliseconds(100));
137 }
138
139 std::cout << "[OBD Thread] Поток завершён. Обработано: " << ok_count << " записей.\n";
140 } catch (const std::exception& e) {
141 std::cerr << "[OBD Thread] КРИТИЧЕСКАЯ ОШИБКА: " << e.what() << "\n";
142 } catch (...) {
143 std::cerr << "[OBD Thread] НЕИЗВЕСТНАЯ ОШИБКА в потоке OBD.\n";
144 }
145}
